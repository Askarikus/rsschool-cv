#### Simultaneous Equations - Three Variables

import numpy

def solve_eq(eq):
    in_mass = [e[:3] for e in eq]
    out_mass = [e[-1] for e in eq]
    m = numpy.array(in_mass)
    v = numpy.array(out_mass)    
    return [round(i) for i in numpy.linalg.solve(m,v)]
